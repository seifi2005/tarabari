# ========================================
# تنظیمات Kavenegar SMS - کپی کنید در .env
# ========================================

# API Key کاوه‌نگار
KAVENEGAR_API_KEY=your_api_key_here

# قالب OTP (برای ورود)
KAVENEGAR_TEMPLATE=hamtayar-otp

# قالب SMS برای مشتری (اختیاری - اگر نباشد از default استفاده می‌شود)
# ⚠️ مهم: اگر می‌خواهید این خط را در .env داشته باشید، باید در کوتیشن قرار بگیرد:
KAVENEGAR_SMS_TEMPLATE_CUSTOMER="خریدار گرامی {customer_name}. سفارش شماره {order_id} در تاریخ {order_register_date} سامانه ترابری برای ارسال ثبت اولیه شد"

# قالب SMS برای ادمین (اختیاری - اگر نباشد از default استفاده می‌شود)
# ⚠️ مهم: اگر می‌خواهید این خط را در .env داشته باشید، باید در کوتیشن قرار بگیرد:
KAVENEGAR_SMS_TEMPLATE_ADMIN="مدیر سیستم. سفارش شماره {order_id} در تاریخ {order_register_date} سامانه ترابری برای ارسال ثبت اولیه شد"

# ========================================

# اگر نمی‌خواهید این قالب‌ها را در .env تنظیم کنید،
# می‌توانید این دو خط را حذف کنید و از مقادیر پیش‌فرض
# در config/services.php استفاده کنید.
